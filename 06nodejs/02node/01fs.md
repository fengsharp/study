# 1.异步读取
fs.readFile() 函数会缓冲整个文件
```js
const fs = require('fs');
fs.readFile("./01fsread.js", (err, data) => {
    if (err) { throw err; }
    console.log(String(data));
});
```

# 2.流的方式读取文件
```js
const fs = require("fs");
const stream = fs.createReadStream('./02fsreadstream.js', {autoClose: false});
stream.on("data", (chunk) => {
    console.log("### receive " + chunk.length + ". ");
    console.log(String(chunk)); // or chunk.toString()
});
stream.on("end", () => {
    console.log("### end");
    stream.close();
});
stream.on("close", () => {
    console.log("### close stream");
});
```

# 3. 写文件
```js
const fs = require("fs");
var data = "hello, test fs.writeFile";
fs.writeFile("./del.txt", data, "utf8", (err) => {
    if (err) throw err;
    console.log('文件已被保存');
});
```

# 4. write by stream
```js
const fs = require("fs");
var data = "hello, test fs.writeFile";

var stream = fs.createWriteStream("./del.txt");
stream.on("finish", () => {
    console.log("### finish");
});
stream.on("close", () => {
    console.log("### close");
})
stream.write(data, (err) => {
    if (err) {
        console.log("### write fail.");
    } else {
        console.log("### write ok.");
    }
});
stream.end();
```
